  ```javascript
  // 检查物资临期情况
  function checkExpiryWarnings() {
      const today = new Date();
      const warningThreshold = 30; // 30天内临期
      const expiringItems = [];

      const rows = document.querySelectorAll('#supplyTableBody tr');
      rows.forEach(row => {
          const expiryCell = row.querySelector('.expiry-cell');
          if (expiryCell) {
              const expiryText = expiryCell.textContent;
              if (expiryText && expiryText !== '长期有效' && expiryText !== '-') {
                  const expiryDate = new Date(expiryText);
                  const diffTime = expiryDate - today;
                  const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

                  const statusCell = row.querySelector('.status-cell');
                  if (diffDays <= 0) {
                      // 已过期
                      expiryCell.style.background = '#f8d7da';
                      expiryCell.style.color = '#721c24';
                      if (statusCell) statusCell.textContent = '已过期';
                      if (statusCell) statusCell.style.color = '#dc3545';
                      expiringItems.push({
                          id: row.cells[0].textContent,
                          name: row.cells[1].textContent,
                          expiry: expiryText,
                          status: '已过期'
                      });
                  } else if (diffDays <= warningThreshold) {
                      // 30天内临期
                      expiryCell.style.background = '#fff3cd';
                      expiryCell.style.color = '#856404';
                      if (statusCell) statusCell.textContent = '临期';
                      if (statusCell) statusCell.style.color = '#856404';
                      expiringItems.push({
                          id: row.cells[0].textContent,
                          name: row.cells[1].textContent,
                          expiry: expiryText,
                          status: '临期（' + diffDays + '天）'
                      });
                  }
              }
          }
      });

      // 显示预警提示
      if (expiringItems.length > 0) {
          const warningDiv = document.getElementById('expiryWarning');
          const messageSpan = document.getElementById('expiryMessage');
          warningDiv.style.display = 'block';

          let message = '以下物资即将过期或已过期：';
          expiringItems.forEach(item => {
              message += ' [' + item.id + ' ' + item.name + ' - ' + item.status + ']';
          });
          messageSpan.textContent = message;
      }
  }
  ```

  2. 实现临期清单导出：
  ```javascript
  // 导出临期物资清单
  function exportExpiringSupplies() {
      const rows = document.querySelectorAll('#supplyTableBody tr');
      let exportContent = '物资临期清单导出\n';
      exportContent += '导出时间：' + new Date().toLocaleString() + '\n\n';
      exportContent += '编号\t名称\t有效期\t状态\n';

      rows.forEach(row => {
          const expiryCell = row.querySelector('.expiry-cell');
          const statusCell = row.querySelector('.status-cell');
          if (expiryCell && (statusCell.textContent === '临期' || statusCell.textContent === '已过期')) {
              exportContent += row.cells[0].textContent + '\t' +
                             row.cells[1].textContent + '\t' +
                             expiryCell.textContent + '\t' +
                             statusCell.textContent + '\n';
          }
      });

      // 创建下载
      const blob = new Blob([exportContent], { type: 'text/plain' });
      const url = URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = '临期物资清单_' + new Date().toISOString().slice(0, 10) + '.txt';
      a.click();
      URL.revokeObjectURL(url);
  }
  ```

  ```javascript
  // 检查24小时未处理的投诉
  function check24HourComplaints() {
      const now = new Date();
      const complaintsToRemind = complaints.filter(complaint => {
          if (complaint.status === '待处理') {
              const submittedTime = new Date(complaint.submittedTime);
              const diffHours = (now - submittedTime) / (1000 * 60 * 60);
              return diffHours >= 24;
          }
          return false;
      });

      if (complaintsToRemind.length > 0) {
          let message = '以下投诉已超过24小时未处理，请及时处理：\n';
          complaintsToRemind.forEach(complaint => {
              message += `${complaint.id} - ${complaint.content}（${complaint.submittedTime}）\n`;
          });

      }
  ```
